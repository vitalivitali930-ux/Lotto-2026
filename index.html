/* Window 2: МОДУЛЬ MONOLITH CORE (VCHF-BATTLE-GAMMA) */ const VCHF_ENGINE = { deck: [], pHand: [], eHand: [], table: [], trump: null, isLocked: false, // ГЕНЕРАЦИЯ ПОЛНОЙ МАТРИЦЫ (36 СФЕР) createDeck() { const suits = ['♠', '♣', '♥', '♦']; const ranks = ['6', '7', '8', '9', '10', 'J', 'Q', 'K', 'A']; const weights = {'6':0, '7':1, '8':2, '9':3, '10':4, 'J':5, 'Q':6, 'K':7, 'A':8}; const colors = {'♠':'#000', '♣':'#000', '♥':'#e30a17', '♦':'#e30a17'}; this.deck = []; suits.forEach(s => { ranks.forEach(r => { this.deck.push({ suit: s, rank: r, weight: weights[r], color: colors[s], id: 'vchf_' + Math.random().toString(36).substr(2, 9) }); }); }); // Хаос-тасовка this.deck.sort(() => Math.random() - 0.5); }, // РИТУАЛ РАЗДАЧИ (18 НА 18) launch() { document.getElementById('main-ui').style.display = 'none'; document.getElementById('game-view').style.display = 'block'; document.getElementById('btn-ritual').style.display = 'block'; this.createDeck(); // Визуализация рубашек в центре const origin = document.getElementById('origin'); origin.innerHTML = ''; this.deck.forEach((card, i) => { const el = document.createElement('div'); el.className = 'card shirt'; el.id = card.id; el.style.transform = `translate(-50%, -50%) scale(0) rotate(${i*10}deg)`; origin.appendChild(el); setTimeout(() => { el.style.transform = `translate(-50%, -50%) scale(1) rotate(${Math.random()*360}deg)`; }, i * 20); }); document.getElementById('btn-ritual').onclick = () => this.ritualStep(); }, ritualStep() { playBeep(220, 0.6, 'sawtooth', 0.05); document.getElementById('btn-ritual').style.display = 'none'; // Разделение массива this.pHand = this.deck.slice(0, 18); this.eHand = this.deck.slice(18, 36); this.trump = this.pHand[0].suit; // Козырь по первой карте Бродяги const tInfo = document.getElementById('trump-info'); tInfo.innerHTML = this.trump; tInfo.style.color = (this.trump === '♥' || this.trump === '♦') ? 'var(--vchf-red)' : 'var(--gold)'; this.deployToLanes(); }, // ГЕОМЕТРИЯ МАГИСТРАЛЕЙ (БУТОН) deployToLanes() { const laneIds = { '♠': 'lane-spades', '♣': 'lane-clubs', '♥': 'lane-hearts', '♦': 'lane-diamonds' }; this.pHand.sort((a,b) => a.weight - b.weight).forEach((card, i) => { const cardEl = document.getElementById(card.id); cardEl.classList.replace('shirt', 'face'); cardEl.innerHTML = `${card.rank}${card.suit}`; cardEl.style.color = card.color; const lane = document.getElementById(laneIds[card.suit]); const suitGroup = this.pHand.filter(c => c.suit === card.suit); const idxInSuit = suitGroup.findIndex(c => c.id === card.id); // Математика веера: исключаем наслоение const total = suitGroup.length; const spread = total > 6 ? 35 : 50; const offset = (idxInSuit - (total - 1) / 2) * spread; const arcY = Math.abs(offset) * 0.15; setTimeout(() => { lane.appendChild(cardEl); cardEl.style.left = `calc(50% + ${offset}px)`; cardEl.style.top = `${arcY}px`; cardEl.style.transform = `translate(-50%, 0) rotate(${offset/10}deg)`; cardEl.style.zIndex = 100 + idxInSuit; cardEl.onclick = () => this.playerAttack(card); }, i * 40); }); }, playerAttack(card) { if (this.isLocked) return; this.isLocked = true; const cardEl = document.getElementById(card.id); this.table.push({ atk: card, def: null }); this.pHand = this.pHand.filter(c => c.id !== card.id); // Анимация в Хаос (Центр) const tilt = (Math.random() - 0.5) * 25; cardEl.style.zIndex = 500 + this.table.length; document.getElementById('drop-zone').appendChild(cardEl); cardEl.style.left = '40%'; cardEl.style.top = '40%'; cardEl.style.transform = `translate(-50%, -50%) rotate(${tilt}deg) scale(1.1)`; playBeep(600, 0.1, 'square', 0.03); setTimeout(() => this.sparkAI(), 800); }, sparkAI() { const currentPair = this.table[this.table.length - 1]; const atk = currentPair.atk; // Логика отбива (Smart-Spark) let candidates = this.eHand.filter(c => (c.suit === atk.suit && c.weight > atk.weight) || (c.suit === this.trump && atk.suit !== this.trump) ); if (candidates.length > 0) { candidates.sort((a,b) => a.weight - b.weight); const def = candidates[0]; currentPair.def = def; this.eHand = this.eHand.filter(c => c.id !== def.id); const defEl = document.getElementById(def.id); defEl.classList.replace('shirt', 'face'); defEl.innerHTML = `${def.rank}${def.suit}`; defEl.style.color = def.color; document.getElementById('drop-zone').appendChild(defEl); defEl.style.zIndex = 600 + this.table.length; } else { // ИИ нечем крыть playBeep(150, 0.5, 'sine', 0.1); } this.isLocked = false; } }; // ГЛОБАЛЬНЫЙ ЗАПУСК function launchMonolith() { VCHF_ENGINE.launch(); }
